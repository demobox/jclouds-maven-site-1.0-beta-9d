47     public static class FutureListener<T> {
48        private final Future<T> future;
49        final ExecutorService executor;
50        private final ExecutionList executionList = new ExecutionList();
51        private final AtomicBoolean hasListeners = new AtomicBoolean(false);
52  
53        static <T> FutureListener<T> create(Future<T> future, ExecutorService executor) {
54           return new FutureListener<T>(future, executor);
55        }
56  
57        private FutureListener(Future<T> future, ExecutorService executor) {
58           this.future = checkNotNull(future, "future");
59           this.executor = checkNotNull(executor, "executor");
60        }
61  
62        public void addListener(Runnable listener, Executor exec) {
63  
64           // When a listener is first added, we run a task that will wait for
65           // the future to finish, and when it is done will run the listeners.
66           if (!hasListeners.get() && hasListeners.compareAndSet(false, true)) {
67              executor.execute(new Runnable() {
68                 /* @Override */
69                 public void run() {
70                    try {
71                       future.get();
72                    } catch (CancellationException e) {
73                       // The task was cancelled, so it is done, run the listeners.
74                    } catch (InterruptedException e) {
75                       // This thread was interrupted. This should never happen, so we
76                       // throw an IllegalStateException.
77                       throw new IllegalStateException("Adapter thread interrupted!", e);
78                    } catch (ExecutionException e) {
79                       // The task caused an exception, so it is done, run the listeners.
80                    }
81                    executionList.run();
82                 }
83              });
84           }
85           executionList.add(listener, exec);
86        }
87  
88        Future<T> getFuture() {
89           return future;
90        }
91  
92        ExecutorService getExecutor() {
93           return executor;
94        }
95     }

120    public static class LazyListenableFutureFunctionAdapter<I, O> extends ForwardingObject implements
121             ListenableFuture<O> {
122       private final FutureListener<I> futureListener;
123       private final Function<? super I, ? extends O> function;
124 
125       static <I, O> LazyListenableFutureFunctionAdapter<I, O> create(Future<I> future,
126                Function<? super I, ? extends O> function, ExecutorService executor) {
127          return new LazyListenableFutureFunctionAdapter<I, O>(future, function, executor);
128       }
129 
130       static <I, O> LazyListenableFutureFunctionAdapter<I, O> create(FutureListener<I> futureListener,
131                Function<? super I, ? extends O> function) {
132          return new LazyListenableFutureFunctionAdapter<I, O>(futureListener, function);
133       }
134 
135       private LazyListenableFutureFunctionAdapter(Future<I> future, Function<? super I, ? extends O> function,
136                ExecutorService executor) {
137          this(FutureListener.create(future, executor), function);
138       }
139 
140       private LazyListenableFutureFunctionAdapter(FutureListener<I> futureListener,
141                Function<? super I, ? extends O> function) {
142          this.futureListener = checkNotNull(futureListener, "futureListener");
143          this.function = checkNotNull(function, "function");
144       }
145 
146       /*
147        * Concurrency detail:
148        * 
149        * <p>To preserve the idempotency of calls to this.get(*) calls to the function are only
150        * applied once. A lock is required to prevent multiple applications of the function. The
151        * calls to future.get(*) are performed outside the lock, as is required to prevent calls to
152        * get(long, TimeUnit) to persist beyond their timeout.
153        * 
154        * <p>Calls to future.get(*) on every call to this.get(*) also provide the cancellation
155        * behavior for this.
156        * 
157        * <p>(Consider: in thread A, call get(), in thread B call get(long, TimeUnit). Thread B may
158        * have to wait for Thread A to finish, which would be unacceptable.)
159        * 
160        * <p>Note that each call to Future<O>.get(*) results in a call to Future<I>.get(*), but the
161        * function is only applied once, so Future<I>.get(*) is assumed to be idempotent.
162        */
163 
164       private final Object lock = new Object();
165       private boolean set = false;
166       private O value = null;
167 
168       @Override
169       public O get() throws InterruptedException, ExecutionException {
170          return apply(futureListener.getFuture().get());
171       }
172 
173       @Override
174       public O get(long timeout, TimeUnit unit) throws InterruptedException, ExecutionException, TimeoutException {
175          return apply(futureListener.getFuture().get(timeout, unit));
176       }
177 
178       private O apply(I raw) {
179          synchronized (lock) {
180             if (!set) {
181                value = function.apply(raw);
182                set = true;
183             }
184             return value;
185          }
186       }
187 
188       @Override
189       public boolean cancel(boolean mayInterruptIfRunning) {
190          return futureListener.getFuture().cancel(mayInterruptIfRunning);
191       }
192 
193       @Override
194       public boolean isCancelled() {
195          return futureListener.getFuture().isCancelled();
196       }
197 
198       @Override
199       public boolean isDone() {
200          return futureListener.getFuture().isDone();
201       }
202 
203       @Override
204       public void addListener(Runnable listener, Executor exec) {
205          futureListener.addListener(listener, exec);
206       }
207 
208       @Override
209       protected Object delegate() {
210          return futureListener.getFuture();
211       }
212 
213    }

215    /**
216     * Just like {@code Futures#compose} except that we check the type of the executorService before
217     * creating the Future. If we are single threaded, invoke the function lazy as opposed to
218     * chaining, so that we don't invoke get() early.
219     */
220    public static <I, O> ListenableFuture<O> compose(Future<I> future, final Function<? super I, ? extends O> function,
221             ExecutorService executorService) {
222       if (future instanceof Futures.ListenableFutureAdapter<?>) {
223          Futures.ListenableFutureAdapter<I> lf = (ListenableFutureAdapter<I>) future;
224          if (lf.futureListener.executor.getClass().isAnnotationPresent(SingleThreaded.class))
225             return Futures.LazyListenableFutureFunctionAdapter.create(
226                      ((org.jclouds.concurrent.Futures.ListenableFutureAdapter<I>) future).futureListener, function);
227          else
228             return com.google.common.util.concurrent.Futures.transform(lf, function, executorService);
229       } else if (executorService.getClass().isAnnotationPresent(SingleThreaded.class)) {
230          return Futures.LazyListenableFutureFunctionAdapter.create(future, function, executorService);
231       } else {
232          return com.google.common.util.concurrent.Futures.transform(Futures.makeListenable(future, executorService),
233                   function, executorService);
234       }
235    }

237    /**
238     * Just like {@code Futures#makeListenable} except that we pass in an executorService.
239     * <p/>
240     * Temporary hack until http://code.google.com/p/guava-libraries/issues/detail?id=317 is fixed.
241     */
242    public static <T> ListenableFuture<T> makeListenable(Future<T> future, ExecutorService executorService) {
243       if (future instanceof ListenableFuture<?>) {
244          return (ListenableFuture<T>) future;
245       }
246       return ListenableFutureAdapter.create(future, executorService);
247    }
